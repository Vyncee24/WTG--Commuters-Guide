//Create Database
CREATE DATABASE wtg_database
USE wtg_database

//Users table
CREATE TABLE users (id INT PRIMARY KEY AUTO_INCREMENT, username VARCHAR(100) NOT NULL, email VARCHAR(150) NOT NULL UNIQUE, password VARCHAR(255) NOT NULL, role ENUM('user', 'admin') DEFAULT 'user', created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP);

//Routes table
CREATE TABLE routes ( id INT PRIMARY KEY AUTO_INCREMENT, from_location VARCHAR(150) NOT NULL, to_location VARCHAR(150) NOT NULL, duration VARCHAR(50) NOT NULL, total_fare VARCHAR(50) NOT NULL, is_popular TINYINT(1) DEFAULT 0);

//Route steps table
CREATE TABLE route_steps (id INT PRIMARY KEY AUTO_INCREMENT, route_id INT, step_number INT NOT NULL, instruction TEXT NOT NULL, FOREIGN KEY (route_id) REFERENCES routes(id));

//Saved routes table (recorded routes)
CREATE TABLE saved_routes ( id INT PRIMARY KEY AUTO_INCREMENT, user_id INT, route_id INT, saved_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP, FOREIGN KEY (user_id) REFERENCES users(id), FOREIGN KEY (route_id) REFERENCES routes(id));

//Search history (saves history of records)
CREATE TABLE search_history (id INT PRIMARY KEY AUTO_INCREMENT, user_id INT, route_id INT, searched_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP, FOREIGN KEY (user_id) REFERENCES users(id), FOREIGN KEY (route_id) REFERENCES routes(id));

//Comments table
CREATE TABLE comments (id INT PRIMARY KEY AUTO_INCREMENT, route_id INT, user_id INT, comment_text TEXT NOT NULL, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP, FOREIGN KEY (route_id) REFERENCES routes(id), FOREIGN KEY (user_id) REFERENCES users(id));


//OLAP SYSTEM FOR WEBSITE (STAR SCHEMA)

//User Dimensions
CREATE TABLE dim_users (user_key INT PRIMARY KEY AUTO_INCREMENT, original_id INT, role VARCHAR(20), account_age_group VARCHAR(50));

//Routes Dimenstions
CREATE TABLE dim_routes (route_key INT PRIMARY KEY AUTO_INCREMENT, original_id INT, from_location VARCHAR(150), to_location VARCHAR(150));

//Date Dimensions
CREATE TABLE dim_date (date_key INT PRIMARY KEY, full_date DATE, day_of_week VARCHAR(15), month VARCHAR(15), semester VARCHAR(20));

//Fact Table
CREATE TABLE fact_searches (search_id INT PRIMARY KEY AUTO_INCREMENT, user_key INT, route_key INT, date_key INT, fare_amount DECIMAL(10,2), FOREIGN KEY (user_key) REFERENCES dim_users(user_key), FOREIGN KEY (route_key) REFERENCES dim_routes(route_key), FOREIGN KEY (date_key) REFERENCES dim_date(date_key));
